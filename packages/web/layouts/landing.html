rill 
# Agent tool-approval workflow
[
  pending: [approve: "ready",   deny: "blocked"],
  ready:   [execute: "done",    revoke: "blocked"],
  blocked: [retry: "pending",   cancel: "done"]
] => $workflow

# Resolve next state from current context
"pending" => $state
"approve" => $action

$workflow.$state.$action
# Result: "ready"
